CREATE OR REPLACE FUNCTION merge() RETURNS trigger LANGUAGE plpgsql AS $$
BEGIN
  INSERT INTO tbl2 VALUES (CONCAT('a',NEW.column1)); 
  RETURN NEW;
END;
$$;

CREATE TRIGGER merge_trg AFTER INSERT ON tbl1 FOR EACH ROW EXECUTE FUNCTION merge();
